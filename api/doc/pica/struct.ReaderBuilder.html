pub fn new() -> ReaderBuilder

Create a new ReaderBuilder for reading PICA+ data.

Example

use pica::{ByteRecord, ReaderBuilder, StringRecord};
use std::error::Error;

fn example() -> Result<(), Box<dyn Error>> {
    let data = "003@ \x1f0123456789\x1e\n";
    let mut reader = ReaderBuilder::new().from_reader(data.as_bytes());
    let records = reader
        .records()
        .map(Result::unwrap)
        .collect::<Vec<StringRecord>>();

    assert_eq!(
        records,
        vec![StringRecord::from_byte_record(ByteRecord::from_bytes(
            "003@ \x1f0123456789\x1e\n".as_bytes()
        )?)?]
    );

    Ok(())
}

pub fn skip_invalid(self, yes: bool) -> Self

Whether to skip invalid records or not.

By default, if an invalid record occurs the reader moves forward until the next valid record was found or the end of file is reached. When this flag is disabled (yes is set to false), the iterator item wil be an pica::Error.

Example

use pica::ReaderBuilder;
use std::error::Error;
use std::io::Cursor;

fn example() -> Result<(), Box<dyn Error>> {
    let data = b"003@ \x1f0123\x1e\n003@ \x1f!456\x1e\n003@ \x1f0789\x1e\n";

    let mut reader = ReaderBuilder::new()
        .skip_invalid(false)
        .from_reader(Cursor::new(data));
    let mut iter = reader.records();

    assert!(iter.next().unwrap().is_ok());
    assert!(iter.next().unwrap().is_err());
    assert!(iter.next().unwrap().is_ok());
    assert!(iter.next().is_none());

    let mut reader = ReaderBuilder::new()
        .skip_invalid(true)
        .from_reader(Cursor::new(data));
    let mut iter = reader.records();

    assert!(iter.next().unwrap().is_ok());
    assert!(iter.next().unwrap().is_ok());
    assert!(iter.next().is_none());

    Ok(())
}
